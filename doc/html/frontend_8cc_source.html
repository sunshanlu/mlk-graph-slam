55int Frontend::estimatePose() {
56 auto algorithm = new g2o::OptimizationAlgorithmLevenberg(
57 std::make_unique<BlockSolverType>(std::make_unique<LinearSolverType>()));
58 g2o::SparseOptimizer optimizer;
59 optimizer.setAlgorithm(algorithm);
60
61 Mat3d K = m_camera->m_K;
62 std::vector<PoseEdge *> edges;
63 std::vector<Feature::Ptr> features;
64
65 // pose vertex
66 auto vertex = new PoseVertex;
67 vertex->setId(0);
68 optimizer.addVertex(vertex);
69
70 // construct edge
71 int idx = 1;
72 for (auto &feature : m_currFrame->getLeftFeatures()) {
73 auto mapPoint = feature->m_mapPoint.lock();
74 if (mapPoint == nullptr)
75 continue;
76 auto edge = new PoseEdge(mapPoint->m_position, K);
77 edge->setId(idx++);
78 edge->setVertex(0, vertex);
79 edge->setInformation(Mat2d::Identity());
80 edge->setMeasurement(feature->m_position);
81 edge->setRobustKernel(new g2o::RobustKernelHuber);
82 edges.push_back(edge);
83 features.push_back(feature);
84 optimizer.addEdge(edge);
85 }
86 int outlierNum = detectOutlier(features, edges, optimizer, vertex);
87 int inlierNum = edges.size() - outlierNum;
88 // set track status
89 if (inlierNum > Config::TRACK_GOOD_TH)
90 m_status = TrackStatus::TRACK_GOOD;
91 else if (inlierNum > Config::TRACK_BAD_TH)
92 m_status = TrackStatus::TRACK_BAD;
93 else
94 m_status = TrackStatus::TRACK_LOST;
95
96 return inlierNum;
97}

99int Frontend::detectOutlier(std::vector<Feature::Ptr> &features, std::vector<PoseEdge *> &edges,
100 g2o::SparseOptimizer &optimizer, PoseVertex *vertex) {
101 int outlierNum;
102 for (int iteration = 0; iteration < Config::OUTLIER_DETECT_ITERATIONS; ++iteration) {
103 outlierNum = 0;
104 vertex->setEstimate(m_currFrame->pose());
105 bool ret = optimizer.initializeOptimization();
106 optimizer.optimize(Config::OPTIMIZE_ITERATIONS);
107
108 for (std::size_t idx = 0; idx < edges.size(); ++idx) {
109 auto &feature = features[idx];
110 auto &edge = edges[idx];
111 auto mapPoint = feature->m_mapPoint.lock();
112 assert(mapPoint != nullptr);
113 if (mapPoint->m_isOutlier)
114 edge->computeError();
115
116 if (edge->chi2() > Config::EDGE_CHI2_TH) {
117 mapPoint->m_isOutlier = true;
118 edge->setLevel(1);
119 ++outlierNum;
120 } else {
121 mapPoint->m_isOutlier = false;
122 edge->setLevel(0);
123 }
124
125 if (iteration == 2)
126 edge->setRobustKernel(nullptr);
127 }
128 }
129
130 // set pose and outlier
131 m_currFrame->pose() = vertex->estimate();
132 for (auto &feature : features) {
133 auto mapPoint = feature->m_mapPoint.lock();
134 assert(mapPoint != nullptr);
135 if (mapPoint->m_isOutlier) {
136 feature->m_mapPoint.reset();
138 mapPoint->m_isOutlier = false;
139 }
140 }
141 return outlierNum;
142}

177bool Frontend::triangulate(const std::vector<SE3d> &poses, const std::vector<Vec2d> &pixelPoints,
178 Vec3d &worldPoint) {
179 MatXd A(2 * poses.size(), 4);
180 for (int idx = 0; idx < poses.size(); ++idx) {
181 const Mat34d T = poses[idx].matrix3x4();
182 const Vec2d &pixelPoint = pixelPoints[idx];
183 Vec3d normPoint;
184 normPoint << pixelPoint.x(), pixelPoint.y(), 1;
185 normPoint = m_camera->m_K.inverse() * normPoint;
186 Mat14d xnExp = normPoint.x() * T.row(2) - T.row(0);
187 Mat14d ynExp = normPoint.y() * T.row(2) - T.row(1);
188
189 A.block<1, 4>(idx * 2, 0) = xnExp;
190 A.block<1, 4>(idx * 2 + 1, 0) = ynExp;
191 }
192 auto svd = A.bdcSvd(Eigen::ComputeThinU | Eigen::ComputeThinV);
193 worldPoint = (svd.matrixV().col(3) / svd.matrixV()(3, 3)).head<3>();
194 if (svd.singularValues()[3] / svd.singularValues()[2] < 1e-2 && worldPoint.z() > 0)
195 return true;
196 return false;
197}
