72void Backend::addVertexAndEdge(std::unordered_map<std::size_t, PoseVertex *> &poseVertexes,
73 std::unordered_map<std::size_t, PointVertex *> &pointVertexes,
74 std::unordered_map<Feature::Ptr, PosePointEdge *> &edges,
75 std::vector<Frame::Ptr> &frames, std::vector<MapPoint::Ptr> &points,
76 g2o::SparseOptimizer &graph) {
77 // 激活关键帧Vertx

89 pointVertex->setId(vertexID++);

92 graph.addVertex(pointVertex);
93 pointVertexes.insert(std::make_pair(point->m_id, pointVertex));
94 }
95
96 // 构造edge
97 int edgeID = 0;
98 for (auto &frame : frames) {
99 auto poseVertex = poseVertexes.at(frame->m_keyID); // 找不到会出现异常
100 for (auto &feature : frame->getLeftFeatures()) {
101 auto mapPoint = feature->m_mapPoint.lock();
102 if (mapPoint == nullptr)
103 continue;
104 auto pointVertex = pointVertexes.at(mapPoint->m_id); // 找不到会出现异常
105 auto edge = new PosePointEdge(m_camera->m_K);
106 auto *kernel = new g2o::RobustKernelHuber;
107 kernel->setDelta(Config::EDGE_CHI2_TH);
108 edge->setId(edgeID++);
109 edge->setVertex(0, poseVertex);
110 edge->setVertex(1, pointVertex);
111 edge->setMeasurement(feature->m_position);
112 edge->setInformation(Mat2d::Identity());
113 edge->setRobustKernel(kernel);
114 graph.addEdge(edge);
115 edges.insert(std::make_pair(feature, edge));
116 }

128double Backend::detectOutlier(g2o::SparseOptimizer &graph,
129 std::unordered_map<Feature::Ptr, PosePointEdge *> &edges) {
130 // outlier点检测
131 int outlierNum = 0, inlierNum = 0;
132 double edgeChTh = Config::EDGE_CHI2_TH;
133 for (int iteration = 0; iteration < Config::OUTLIER_DETECT_ITERATIONS; ++iteration) {
134 outlierNum = 0;
135 inlierNum = 0;
136 for (auto iter = edges.begin(); iter != edges.end(); ++iter) {
137 MapPoint::Ptr point = iter->first->m_mapPoint.lock();
138 assert(point != nullptr);
139 PosePointEdge *edge = iter->second;
140 if (edge->chi2() > edgeChTh)
141 ++outlierNum;
142 else
143 ++inlierNum;
144 }
145 if (inlierNum / (outlierNum + inlierNum) > 0.5)
146 break;
147 edgeChTh *= 1.1;
148 }
149
150 // 进行外点剔除（移除Feature和MapPoint之间的关系）
151 for (auto iter = edges.begin(); iter != edges.end(); ++iter) {
152 Feature::Ptr feature = iter->first;
153 MapPoint::Ptr point = feature->m_mapPoint.lock();
154 PosePointEdge *edge = iter->second;
155 assert(point != nullptr);
156 if (edge->chi2() > edgeChTh) {
157 feature->m_mapPoint.reset();
158 point->m_featureNum--;
159 }
160 }
161 return inlierNum / (outlierNum + inlierNum);
162}
